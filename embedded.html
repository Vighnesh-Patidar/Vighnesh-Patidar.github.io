bool mutex_acquire(mutex_t *mutex, uint32_t timeout_ms) {
    // Critical section entry with ISR masking
    CRITICAL_SECTION_ENTER();
    
    task_t *current_task = get_current_task();
    
    if (mutex->owner == NULL) {
        // Mutex is free, acquire it
        mutex->owner = current_task;
        mutex->original_priority = current_task->priority;
        CRITICAL_SECTION_EXIT();
        return true;
    } else if (mutex->owner == current_task) {
        // Recursive lock, increase count
        mutex->lock_count++;
        CRITICAL_SECTION_EXIT();
        return true;
    } else {
        // Mutex is taken, apply priority inheritance
        if (current_task->priority > mutex->owner->priority) {
            // Boost owner's priority to prevent inversion
            task_set_priority(mutex->owner, current_task->priority);
        }
        
        // Add task to mutex wait list
        task_add_to_wait_list(&mutex->wait_list, current_task);
        
        CRITICAL_SECTION_EXIT();
        
        // Wait for mutex with timeout
        return task_wait_for_notification(timeout_ms);
    }
}
